const SubscriptionModal = ({ open, onClose, onConfirm, companyName }) => {
  const [selectedPlan, setSelectedPlan] = useState('trial');
  if (!open) return null;

  const plans = [
    { id:'trial',   name:'Essai gratuit',       price:'0',   period:'30 jours', desc:'Toutes les fonctionnalités pendant 30 jours, sans carte bancaire' },
    { id:'monthly', name:'Abonnement mensuel',   price:'49',  period:'mois',     desc:'Paiement mensuel, résiliable à tout moment' },
    { id:'yearly',  name:'Abonnement annuel',    price:'499', period:'an',       desc:'Économisez 15% avec l\'engagement annuel' },
  ];

  return (
    <div className="ta-modal-overlay" onClick={onClose}>
      <div className="ta-modal" onClick={e => e.stopPropagation()}>
        <div className="ta-modal-head">
          <div className="icon">🎉</div>
          <h3>Choisissez votre formule</h3>
          <p>{companyName || "Votre entreprise"}, sélectionnez l'offre qui vous convient</p>
        </div>
        {plans.map(plan => (
          <div key={plan.id} className={`ta-plan-option${selectedPlan===plan.id?' selected':''}`} onClick={() => setSelectedPlan(plan.id)}>
            <div>
              <div className="ta-plan-name">{plan.name}</div>
              <div className="ta-plan-desc">{plan.desc}</div>
            </div>
            <div className="ta-plan-price">
              {plan.price==='0' ? 'Gratuit' : `${plan.price}€`}
              <small> / {plan.period}</small>
            </div>
          </div>
        ))}
        <div className="ta-modal-actions">
          <button className="ta-modal-btn ta-modal-btn-secondary" onClick={onClose}>Annuler</button>
          <button className="ta-modal-btn ta-modal-btn-primary" onClick={() => onConfirm(selectedPlan)}>
            Paiement en ligne →
          </button>
        </div>
      </div>
    </div>
  );
};

  const [showRegPwd, setShowRegPwd] = useState(false);
  const [regMsg,     setRegMsg]     = useState({ type:'', text:'' });
  const [subOpen,    setSubOpen]    = useState(false);

  const companies = [
    { id:'1', name:'Transport Express Sarl' },
    { id:'2', name:'Mega Logistique SA'     },
    { id:'3', name:'City Move Services'     },
  ];

  const dispatch  = useDispatch();
  const navigate  = useNavigate();

  useEffect(() => {
    const saved = localStorage.getItem('rememberedEmail');
    if (saved) { setEmail(saved); setRememberMe(true); }
  }, []);

  const handleLogin = async e => {
    e.preventDefault();
    if (!email || !password)    { setMsg({ type:'error', text:'Veuillez remplir tous les champs.' }); return; }
    if (!selectedCompany)       { setMsg({ type:'error', text:'Veuillez sélectionner votre entreprise.' }); return; }
    setLoading(true); setMsg({ type:'', text:'' });
    try {
      toast.success(`Bienvenue chez ${companies.find(c=>c.id===selectedCompany)?.name} 👋`);
      localStorage.setItem('adminToken', '12345');
      if (rememberMe) localStorage.setItem('rememberedEmail', email);
      else            localStorage.removeItem('rememberedEmail');
      navigate('/dashboard');
    } catch {
      setMsg({ type:'error', text:'Email, mot de passe ou entreprise incorrect.' });
    } finally {
      setLoading(false);
    }
  };

  const handleRegister = async e => {
    e.preventDefault();
    if (!reg.companyName || !reg.email || !reg.password || !reg.confirm) {
      setRegMsg({ type:'error', text:'Tous les champs sont obligatoires.' }); return;
    }
    if (reg.password !== reg.confirm) {
      setRegMsg({ type:'error', text:'Les mots de passe ne correspondent pas.' }); return;
    }
    if (reg.password.length < 8) {
      setRegMsg({ type:'error', text:'Mot de passe trop court (min. 8 caractères).' }); return;
    }
    setLoading(true); setRegMsg({ type:'', text:'' });
    try {
      await new Promise(r => setTimeout(r, 1100));
      toast.success('Entreprise créée !');
      setSubOpen(true);
    } catch {
      setRegMsg({ type:'error', text:"Erreur lors de la création de l'entreprise." });
    } finally {
      setLoading(false);
    }
  };

  const handleSubConfirm = async (plan) => {
    setSubOpen(false);
    toast.info(`Redirection vers le paiement — offre ${plan}...`);
    await new Promise(r => setTimeout(r, 1400));
    toast.success('Inscription finalisée — connectez-vous !');
    setMode('login'); setEmail(reg.email);
    setMsg({ type:'success', text:'Entreprise créée et abonnement souscrit — connectez-vous ci-dessous.' });
    setReg({ companyName:'', email:'', password:'', confirm:'' });
  };

  const socialMsg = p => toast.info(`${p} OAuth — intégration à venir`);

  return (
    <>
      <style>{css}</style>
      <div className="ta-root">
        <div className="ta-orb ta-orb1"/>
        <div className="ta-orb ta-orb2"/>
        <div className="ta-orb ta-orb3"/>

        <div className="ta-card">

          {/* ── Colonne gauche ── */}
          <div className="ta-left">
            <div className="ta-logo-row">
              <div className="ta-logo-mark"><IconLayers/></div>
              <div>
                <div className="ta-logo-name">SiRA<span style={{ color:'var(--accent)' }}>.</span></div>
                <div className="ta-logo-sub">OPS Center</div>
              </div>
            </div>

            <div className="ta-hero">
              <div className="ta-eyebrow">
                <div className="ta-eyebrow-dot"/>
                Panneau de contrôle
              </div>
              <h1 className="ta-title">
                Gérez vos<br/>courses avec<br/><span>précision</span>
              </h1>
              <p className="ta-desc">
                Plateforme de pilotage complète pour les opérateurs de livraison modernes.
              </p>
            </div>

            <div className="ta-features">
              {[
                { icon:"🛵", text:"Suivi temps réel des conducteurs"     },
                { icon:"🔍", text:"Détection de fraude intelligente"      },
                { icon:"📦", text:"Gestion des colis multi-critères"      },
                { icon:"📊", text:"Tableaux de bord analytiques"          },
              ].map((f, i) => (
                <div className="ta-feat" key={i}>
                  <div className="ta-feat-icon">{f.icon}</div>
                  <span className="ta-feat-text">{f.text}</span>
                </div>
              ))}
            </div>

            <div className="ta-status">
              <div className="ta-pulse"/>
              Système opérationnel · v2.0.0
            </div>
          </div>

          {/* ── Colonne droite ── */}
          <div className="ta-right">
            <div className="ta-form-title">
              {mode === 'login' ? 'Bon retour 👋' : 'Créer votre espace'}
            </div>
            <div className="ta-form-sub">
              {mode === 'login'
                ? 'Connectez-vous à votre tableau de bord administrateur'
                : 'Inscrivez votre entreprise sur la plateforme SiRA'}
            </div>

            {/* Tabs */}
            <div className="ta-tabs">
              <button className={`ta-tab${mode==='login'?' active':''}`} onClick={() => { setMode('login'); setMsg({type:'',text:''}); }}>Connexion</button>
              <button className={`ta-tab${mode==='register'?' active':''}`} onClick={() => { setMode('register'); setRegMsg({type:'',text:''}); }}>Inscription</button>
            </div>

            {/* ── Formulaire connexion ── */}
            {mode === 'login' && (
              <form onSubmit={handleLogin}>
                <Msg {...msg}/>
                <FieldInput label="Entreprise" value={selectedCompany} onChange={setSelectedCompany} icon={<IconBuilding/>} options={companies}/>
                <FieldInput label="Adresse e-mail" type="email" value={email} onChange={setEmail} placeholder="vous@entreprise.com" icon={<IconMail/>}/>
                <FieldInput label="Mot de passe" type={showPwd?'text':'password'} value={password} onChange={setPassword} placeholder="••••••••••" icon={<IconLock/>} showEye eyeOpen={showPwd} onToggleEye={() => setShowPwd(v=>!v)}/>
                <div className="ta-row">
                  <label className="ta-remember" onClick={() => setRememberMe(v=>!v)}>
                    <div className={`ta-check${rememberMe?' on':''}`}><IconCheck/></div>
                    <span>Se souvenir de moi</span>
                  </label>
                  <button type="button" className="ta-forgot">Mot de passe oublié ?</button>
                </div>
                <button type="submit" className="ta-btn" disabled={loading}>
                  {loading ? <div className="ta-spinner"/> : <>Se connecter <IconArrow/></>}
                </button>
                <div className="ta-sep">
                  <div className="ta-sep-line"/><span className="ta-sep-label">ou continuer avec</span><div className="ta-sep-line"/>
                </div>
                <div className="ta-socials">
                  <button type="button" className="ta-soc-btn" onClick={() => socialMsg('Google')}><GoogleLogo/> Google</button>
                  <button type="button" className="ta-soc-btn" onClick={() => socialMsg('Microsoft')}><MicrosoftLogo/> Microsoft</button>
                </div>
              </form>
            )}

            {/* ── Formulaire inscription ── */}
            {mode === 'register' && (
              <form onSubmit={handleRegister}>
                <Msg {...regMsg}/>
                <FieldInput label="Nom de l'entreprise" value={reg.companyName} onChange={v => setReg(r=>({...r,companyName:v}))} placeholder="Ma Société SARL" icon={<IconBuilding/>}/>
                <FieldInput label="E-mail professionnel" type="email" value={reg.email} onChange={v => setReg(r=>({...r,email:v}))} placeholder="contact@entreprise.com" icon={<IconMail/>}/>
                <FieldInput label="Mot de passe" type={showRegPwd?'text':'password'} value={reg.password} onChange={v => setReg(r=>({...r,password:v}))} placeholder="Min. 8 caractères" icon={<IconLock/>} showEye eyeOpen={showRegPwd} onToggleEye={() => setShowRegPwd(v=>!v)}/>
                <FieldInput label="Confirmer le mot de passe" type="password" value={reg.confirm} onChange={v => setReg(r=>({...r,confirm:v}))} placeholder="Répétez le mot de passe" icon={<IconShield/>}/>
                <button type="submit" className="ta-btn" disabled={loading} style={{ marginTop:6 }}>
                  {loading ? <div className="ta-spinner"/> : <>Créer mon entreprise <IconAddUser/></>}
                </button>
                <div className="ta-switch">
                  Déjà un compte ? <button type="button" onClick={() => setMode('login')}>Se connecter</button>
                </div>
              </form>
            )}
          </div>
        </div>
      </div>

      <SubscriptionModal
        open={subOpen}
        onClose={() => setSubOpen(false)}
        onConfirm={handleSubConfirm}
        companyName={reg.companyName}
      />
    </>
  );
}

export default LoginPage;
