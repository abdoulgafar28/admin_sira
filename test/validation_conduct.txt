  "En Attente": { bg:T.amberPl, bd:T.amberBd, color:T.amber   },

function initiales(nom) {
  return nom.split(" ").map(w => w[0]).join("").slice(0,2).toUpperCase();
}


function Av({ nom, size = 36 }) {
  const ini = initiales(nom);
  const p   = avColor(ini);
  return (
    <div style={{
      width:size, height:size, borderRadius:"50%", flexShrink:0,
      background:p.bg, color:p.c, border:`1.5px solid ${p.c}30`,
      display:"flex", alignItems:"center", justifyContent:"center",
      fontSize:size * 0.35, fontWeight:700, fontFamily:T.mono,
    }}>{ini}</div>
  );
}

function Tag({ label, bg, bd, color }) {
  return (
    <span style={{
      display:"inline-block", padding:"4px 12px", borderRadius:20,
      fontSize:11, fontWeight:600, fontFamily:T.sans,
      background:bg, color, border:`1px solid ${bd}`,
    }}>{label}</span>
  );
}


function DocPlaceholder({ type }) {
  const configs = {
    cni:        { bg:"#e8f5e9", accent:"#81c784", label:"CNI",   lines:[22,18,16] },
    permis:     { bg:"#e3f2fd", accent:"#64b5f6", label:"PERM",  lines:[20,16,14] },
    carteGrise: { bg:"#fff8e1", accent:"#ffd54f", label:"C.G.",  lines:[22,18,16] },
    assurance:  { bg:"#fce4ec", accent:"#f48fb1", label:"ASS.",  lines:[20,16,14] },
    photo:      { bg:"#f3e5f5", accent:"#ce93d8", label:"PHOTO", lines:[] },
    moto:       { bg:"#e8eaf6", accent:"#9fa8da", label:"MOTO",  lines:[] },
  };
  const c = configs[type] || configs.cni;
  return (
    <div style={{
      width:"100%", height:90, borderRadius:8, background:c.bg,
      display:"flex", flexDirection:"column", alignItems:"center", justifyContent:"center",
      gap:5, overflow:"hidden",
    }}>
      {type === "photo" && (
        <>
          <div style={{ width:30, height:30, borderRadius:"50%", background:c.accent, opacity:.7 }} />
          <div style={{ width:50, height:20, borderRadius:"8px 8px 0 0", background:c.accent, opacity:.5 }} />
        </>
      )}
      {type === "moto" && (
        <div style={{ display:"flex", gap:6, alignItems:"flex-end" }}>
          <div style={{ width:14, height:14, borderRadius:"50%", background:c.accent, opacity:.6 }} />
          <div style={{ width:40, height:16, borderRadius:4, background:c.accent, opacity:.5 }} />
          <div style={{ width:14, height:14, borderRadius:"50%", background:c.accent, opacity:.6 }} />
        </div>
      )}
      {type !== "photo" && type !== "moto" && (
        <>
          <div style={{ fontFamily:T.mono, fontSize:10, fontWeight:700, color:c.accent, letterSpacing:1, opacity:.9 }}>{c.label}</div>
          {c.lines.map((w, i) => (
            <div key={i} style={{ width:w*3, height:3, borderRadius:2, background:c.accent, opacity:.35 }} />

      )}
    </div>
  );
