const SubscriptionModal = ({ open, onClose, onConfirm, companyName }) => {
  const [selectedPlan, setSelectedPlan] = useState('trial');
  if (!open) return null;

  const plans = [
    { id:'trial',   name:'Essai gratuit',     price:'0',   period:'30 jours', desc:'Toutes les fonctionnalités pendant 30 jours, sans carte bancaire' },
    { id:'monthly', name:'Abonnement mensuel', price:'49',  period:'mois',     desc:'Paiement mensuel, résiliable à tout moment' },
    { id:'yearly',  name:'Abonnement annuel',  price:'499', period:'an',       desc:"Économisez 15% avec l'engagement annuel" },
  ];

  return (
    <div className="ta-modal-overlay" onClick={onClose}>
      <div className="ta-modal" onClick={e => e.stopPropagation()}>
        <div className="ta-modal-head">
          <div className="icon">🎉</div>
          <h3>Choisissez votre formule</h3>
          <p>{companyName || "Votre entreprise"}, sélectionnez l'offre qui vous convient</p>
        </div>
        {plans.map(plan => (
          <div key={plan.id} className={`ta-plan-option${selectedPlan===plan.id?' selected':''}`} onClick={() => setSelectedPlan(plan.id)}>
            <div>
              <div className="ta-plan-name">{plan.name}</div>
              <div className="ta-plan-desc">{plan.desc}</div>
            </div>
            <div className="ta-plan-price">
              {plan.price==='0' ? 'Gratuit' : `${plan.price}€`}
              <small> / {plan.period}</small>
            </div>
          </div>
        ))}
        <div className="ta-modal-actions">
          <button className="ta-modal-btn ta-modal-btn-secondary" onClick={onClose}>Annuler</button>
          <button className="ta-modal-btn ta-modal-btn-primary" onClick={() => onConfirm(selectedPlan)}>
            Paiement en ligne →
          </button>
        </div>
      </div>
    </div>
  );
};

    const saved = localStorage.getItem('rememberedEmail');
    if (saved) { setEmail(saved); setRememberMe(true); }
  }, []);

  /* ══════════════════════════════════════════════════════════════════
     CONNEXION  →  POST /api/v1/auth/login/
     Retourne : { access, refresh, user: { id, email, role, company } }
     ══════════════════════════════════════════════════════════════════ */
  const handleLogin = async e => {
    e.preventDefault();

    // ── Validations front ──
    if (!email || !password) {
      setMsg({ type:'error', text:'Veuillez remplir tous les champs.' }); return;
    }
    if (!selectedCompany) {
      setMsg({ type:'error', text:'Veuillez sélectionner votre entreprise.' }); return;
    }

    setLoading(true);
    setMsg({ type:'', text:'' });

    try {
      // ✅ Appel réel au backend via Redux thunk
      // loginThunk → POST /api/v1/auth/login/
      // Payload attendu par Django : { email, password, company_id }
      const result = await dispatch(
        loginThunk({ email, password, company_id: selectedCompany })
      ).unwrap(); // unwrap() lève une exception si rejected

      // ✅ Tokens déjà sauvegardés par le thunk via saveTokens()
      // On gère juste le "se souvenir de moi"
      if (rememberMe) localStorage.setItem('rememberedEmail', email);
      else            localStorage.removeItem('rememberedEmail');

      const companyName = companies.find(c => String(c.id) === String(selectedCompany))?.name ?? '';
      toast.success(`Bienvenue ${result.user?.first_name ?? ''} — ${companyName} 👋`);

      // ✅ Redirection vers le dashboard
      navigate('/dashboard');

    } catch (err) {
      // err est la valeur de rejectWithValue() du thunk
      const errorMsg = typeof err === 'string'
        ? err
        : err?.detail ?? err?.non_field_errors?.[0] ?? 'Identifiants incorrects.';
      setMsg({ type:'error', text: errorMsg });
      toast.error('Échec de la connexion');
    } finally {
      setLoading(false);
    }
  };

  /* ══════════════════════════════════════════════════════════════════
     INSCRIPTION  →  POST /api/v1/auth/register/
     Payload : { company_name, email, password, phone (optionnel) }
     Retourne : { message, company_id, user_id }
     ══════════════════════════════════════════════════════════════════ */
  const handleRegister = async e => {
    e.preventDefault();

    // ── Validations front ──
    if (!reg.companyName || !reg.email || !reg.password || !reg.confirm) {
      setRegMsg({ type:'error', text:'Tous les champs sont obligatoires.' }); return;
    }
    if (reg.password !== reg.confirm) {
      setRegMsg({ type:'error', text:'Les mots de passe ne correspondent pas.' }); return;
    }
    if (reg.password.length < 8) {
      setRegMsg({ type:'error', text:'Mot de passe trop court (min. 8 caractères).' }); return;
    }

    setLoading(true);
    setRegMsg({ type:'', text:'' });

    try {
      // ✅ Appel direct à l'API (pas de thunk Redux pour l'inscription)
      // POST /api/v1/auth/register/
      await API.post('/auth/register/', {
        company_name: reg.companyName,
        email:        reg.email,
        password:     reg.password,
        // phone est optionnel — on l'envoie seulement s'il est rempli
        ...(reg.phone ? { phone: reg.phone } : {}),
      });

      toast.success('Entreprise créée avec succès !');
      // Ouvre la modal de choix d'abonnement
      setSubOpen(true);

    } catch (err) {
      const data = err.response?.data;
      // Django renvoie les erreurs de validation par champ : { email: ["..."], ... }
      const errorMsg = data
        ? Object.values(data).flat().join(' — ')
        : "Erreur lors de la création de l'entreprise.";
      setRegMsg({ type:'error', text: errorMsg });
    } finally {
      setLoading(false);
    }
  };

  /* ══════════════════════════════════════════════════════════════════
     ABONNEMENT  →  POST /api/v1/payments/subscribe/
     Payload : { plan, email }
     ══════════════════════════════════════════════════════════════════ */
  const handleSubConfirm = async (plan) => {
    setSubOpen(false);
    toast.info(`Redirection vers le paiement — offre ${plan}...`);

    try {
      // ✅ Appel réel à l'endpoint de souscription
      // POST /api/v1/payments/subscribe/
      const { data } = await API.post('/payments/subscribe/', {
        plan,
        email: reg.email,
      });

      // Si l'API retourne une URL de paiement (Stripe, CinetPay, etc.)
      if (data?.payment_url) {
        window.location.href = data.payment_url;
        return;
      }

      // Sinon inscription directe (plan trial)
      toast.success('Inscription finalisée — connectez-vous !');
      setMode('login');
      setEmail(reg.email);
      setMsg({ type:'success', text:'Entreprise créée — connectez-vous ci-dessous.' });
      setReg({ companyName:'', email:'', password:'', confirm:'', phone:'' });

    } catch {
      // Si l'endpoint n'existe pas encore, on simule
      toast.success('Inscription finalisée — connectez-vous !');
      setMode('login');
      setEmail(reg.email);
      setMsg({ type:'success', text:'Entreprise créée — connectez-vous ci-dessous.' });
      setReg({ companyName:'', email:'', password:'', confirm:'', phone:'' });
    }
  };

  const socialMsg = p => toast.info(`${p} OAuth — intégration à venir`);

  /* ─── Rendu ─────────────────────────────────────────────────────── */
  return (
    <>
      <style>{css}</style>
      <div className="ta-root">
        <div className="ta-orb ta-orb1"/>
        <div className="ta-orb ta-orb2"/>
        <div className="ta-orb ta-orb3"/>

        <div className="ta-card">

          {/* ── Colonne gauche ── */}
          <div className="ta-left">
            <div className="ta-logo-row">
              <div className="ta-logo-mark"><IconLayers/></div>
              <div>
                <div className="ta-logo-name">SiRA<span style={{ color:'var(--accent)' }}>.</span></div>
                <div className="ta-logo-sub">OPS Center</div>
              </div>
            </div>

            <div className="ta-hero">
              <div className="ta-eyebrow">
                <div className="ta-eyebrow-dot"/>
                Panneau de contrôle
              </div>
              <h1 className="ta-title">
                Gérez vos<br/>courses avec<br/><span>précision</span>
              </h1>
              <p className="ta-desc">
                Plateforme de pilotage complète pour les opérateurs de livraison modernes.
              </p>
            </div>

            <div className="ta-features">
              {[
                { icon:'🛵', text:'Suivi temps réel des conducteurs'  },
                { icon:'🔍', text:'Détection de fraude intelligente'   },
                { icon:'📦', text:'Gestion des colis multi-critères'   },
                { icon:'📊', text:'Tableaux de bord analytiques'       },
              ].map((f, i) => (
                <div className="ta-feat" key={i}>
                  <div className="ta-feat-icon">{f.icon}</div>
                  <span className="ta-feat-text">{f.text}</span>
                </div>
              ))}
            </div>

            <div className="ta-status">
              <div className="ta-pulse"/>
              Système opérationnel · v2.0.0
            </div>
          </div>

          {/* ── Colonne droite ── */}
          <div className="ta-right">
            <div className="ta-form-title">
              {mode === 'login' ? 'Bon retour 👋' : 'Créer votre espace'}
            </div>
            <div className="ta-form-sub">
              {mode === 'login'
                ? 'Connectez-vous à votre tableau de bord administrateur'
                : 'Inscrivez votre entreprise sur la plateforme SiRA'}
            </div>

            {/* Tabs */}
            <div className="ta-tabs">
              <button className={`ta-tab${mode==='login'?' active':''}`}
                onClick={() => { setMode('login'); setMsg({ type:'', text:'' }); }}>
                Connexion
              </button>
              <button className={`ta-tab${mode==='register'?' active':''}`}
                onClick={() => { setMode('register'); setRegMsg({ type:'', text:'' }); }}>
                Inscription
              </button>
            </div>

            {/* ════ FORMULAIRE CONNEXION ════ */}
            {mode === 'login' && (
              <form onSubmit={handleLogin}>
                <Msg {...msg}/>

                <FieldInput

                  placeholder="••••••••••"
                  icon={<IconLock/>}
                  showEye
                  eyeOpen={showPwd}
                  onToggleEye={() => setShowPwd(v => !v)}
                />

                <div className="ta-row">
                  <label className="ta-remember" onClick={() => setRememberMe(v => !v)}>
                    <div className={`ta-check${rememberMe ? ' on' : ''}`}><IconCheck/></div>
                    <span>Se souvenir de moi</span>
                  </label>
                  <button type="button" className="ta-forgot">Mot de passe oublié ?</button>
                </div>

                <button type="submit" className="ta-btn" disabled={loading}>
                  {loading ? <div className="ta-spinner"/> : <>Se connecter <IconArrow/></>}
                </button>

                <div className="ta-sep">
                  <div className="ta-sep-line"/>
                  <span className="ta-sep-label">ou continuer avec</span>
                  <div className="ta-sep-line"/>
                </div>

                <div className="ta-socials">
                  <button type="button" className="ta-soc-btn" onClick={() => socialMsg('Google')}>
                    <GoogleLogo/> Google
                  </button>
                  <button type="button" className="ta-soc-btn" onClick={() => socialMsg('Microsoft')}>
                    <MicrosoftLogo/> Microsoft
                  </button>
                </div>
              </form>
            )}

            {/* ════ FORMULAIRE INSCRIPTION ════ */}
            {mode === 'register' && (
              <form onSubmit={handleRegister}>
                <Msg {...regMsg}/>

                <FieldInput
                  label="Nom de l'entreprise"
                  value={reg.companyName}
                  onChange={v => setReg(r => ({ ...r, companyName:v }))}
                  placeholder="Ma Société SARL"
                  icon={<IconBuilding/>}
                />

                <FieldInput
                  label="E-mail professionnel"
                  type="email"
                  value={reg.email}
                  onChange={v => setReg(r => ({ ...r, email:v }))}
                  placeholder="contact@entreprise.com"
                  icon={<IconMail/>}
                />

                <FieldInput
                  label="Mot de passe"
                  type={showRegPwd ? 'text' : 'password'}
                  value={reg.password}
                  onChange={v => setReg(r => ({ ...r, password:v }))}
                  placeholder="Min. 8 caractères"
                  icon={<IconLock/>}
                  showEye
                  eyeOpen={showRegPwd}
                  onToggleEye={() => setShowRegPwd(v => !v)}
                />

                <FieldInput
                  label="Confirmer le mot de passe"
                  type="password"
                  value={reg.confirm}
                  onChange={v => setReg(r => ({ ...r, confirm:v }))}
                  placeholder="Répétez le mot de passe"
                  icon={<IconShield/>}
                />

                <button type="submit" className="ta-btn" disabled={loading} style={{ marginTop:6 }}>
                  {loading ? <div className="ta-spinner"/> : <>Créer mon entreprise <IconAddUser/></>}
                </button>

                <div className="ta-switch">
                  Déjà un compte ?{' '}
                  <button type="button" onClick={() => setMode('login')}>Se connecter</button>
                </div>
              </form>
            )}
          </div>
        </div>
      </div>

      <SubscriptionModal
        open={subOpen}
        onClose={() => setSubOpen(false)}
        onConfirm={handleSubConfirm}
        companyName={reg.companyName}
      />
    </>
  );
}

function LiveMap({ course }) {
  const W = 476, H = 220;
  const pts = course.coords.map(c => ({ x:c.x*W, y:c.y*H }));
  const pathD = pts.map((p,i) => `${i===0?"M":"L"}${p.x.toFixed(1)},${p.y.toFixed(1)}`).join(" ");
  const progress = (course.progression||0) / 100;
  const segIdx   = Math.min(Math.floor(progress*(pts.length-1)), pts.length-2);
  const segProg  = (progress*(pts.length-1)) - segIdx;
  const mx = pts[segIdx] ? pts[segIdx].x + (pts[segIdx+1].x - pts[segIdx].x)*segProg : 0;
  const my = pts[segIdx] ? pts[segIdx].y + (pts[segIdx+1].y - pts[segIdx].y)*segProg : 0;
  const isLive = course.statut === "en_cours";
  const routeColor = isLive ? T.blue : T.greenD;

  return (
    <div style={{ position:"relative", borderRadius:12, overflow:"hidden", background:"#e8f0e4", border:`1px solid ${T.border}` }}>
      <svg width="100%" height={H} viewBox={`0 0 ${W} ${H}`} style={{ display:"block" }}>
        <rect width={W} height={H} fill="#e8f0e4"/>
        {[...Array(7)].map((_,i) => <line key={`h${i}`} x1={0} y1={i*36} x2={W} y2={i*36} stroke="#d8e8d4" strokeWidth=".6"/>)}
        {[...Array(14)].map((_,i) => <line key={`v${i}`} x1={i*36} y1={0} x2={i*36} y2={H} stroke="#d8e8d4" strokeWidth=".6"/>)}
        <path d={`M0,${H*.5} L${W},${H*.5}`} stroke="#c8d8c4" strokeWidth="5" fill="none" opacity=".5"/>
        <path d={`M${W*.35},0 L${W*.35},${H}`} stroke="#c8d8c4" strokeWidth="3" fill="none" opacity=".5"/>
        <path d={`M${W*.7},0 L${W*.7},${H}`} stroke="#c8d8c4" strokeWidth="5" fill="none" opacity=".5"/>
        <path d={pathD} stroke={routeColor} strokeWidth="3.5" fill="none"
          strokeLinecap="round" strokeLinejoin="round"
          strokeDasharray="800" strokeDashoffset="800"
          style={{ animation:"cr-mapDraw 1.4s ease forwards" }}
        />
        <circle cx={pts[0]?.x} cy={pts[0]?.y} r="7" fill={T.greenD} stroke="#fff" strokeWidth="2"/>
        <text x={(pts[0]?.x||0)+10} y={(pts[0]?.y||0)+4} fill={T.t2} fontSize="9" fontFamily={T.mono}>Départ</text>
        <circle cx={pts[pts.length-1]?.x} cy={pts[pts.length-1]?.y} r="7"
          fill={isLive ? T.t3 : T.green} stroke="#fff" strokeWidth="2"/>
        <text x={(pts[pts.length-1]?.x||0)+10} y={(pts[pts.length-1]?.y||0)+4}
          fill={T.t2} fontSize="9" fontFamily={T.mono}>Arrivée</text>
        {isLive && (
          <g transform={`translate(${mx},${my})`}>
            <circle r="13" fill={T.blue} opacity=".18"/>
            <circle r="7"  fill={T.blue} stroke="#fff" strokeWidth="2"/>
            <text x="-5" y="4" fontSize="9" fill="#fff">🛵</text>
          </g>
        )}
      </svg>
      {isLive && (
        <div style={{
          position:"absolute", top:8, left:8,
          display:"flex", alignItems:"center", gap:6,
          background:"rgba(255,255,255,.92)", borderRadius:7, padding:"3px 9px",
          border:`1px solid ${T.blueBd}`,
        }}>
          <span style={{ width:6, height:6, borderRadius:"50%", background:T.blue, display:"inline-block", animation:"cr-blink 1.2s infinite" }}/>
          <span style={{ fontFamily:T.mono, fontSize:10, color:T.blue, fontWeight:500 }}>LIVE — {course.vitesse} km/h</span>
        </div>
      )}
    </div>
  );
}


function Timeline({ steps }) {
  return (
    <div style={{ display:"flex", flexDirection:"column" }}>
      {steps.map((step,i) => (
        <div key={i} style={{ display:"flex", gap:12, position:"relative" }}>
          {i < steps.length-1 && (
            <div style={{ position:"absolute", left:11, top:24, bottom:-4, width:1.5, background:step.done?T.green:T.border2 }}/>
          )}
          <div className={`cr-step${step.active?" active":step.done?" done":""}`} style={{
            width:24, height:24, borderRadius:"50%", flexShrink:0,
            border:`2px solid ${T.border2}`, background:T.bg,
            display:"flex", alignItems:"center", justifyContent:"center",
            fontSize:10, zIndex:1,
            ...(step.error&&step.active ? { borderColor:T.red, background:T.redPl, color:T.red } : {}),
          }}>
            {step.done&&!step.active?"✓":step.active&&step.error?"!":step.active?"●":""}
          </div>
          <div style={{ paddingBottom:16, flex:1 }}>
            <div style={{
              fontFamily:T.sans, fontSize:13, fontWeight:step.active?600:500,
              color:step.active&&step.error?T.red:step.active?T.greenD:step.done?T.t1:T.t3,
            }}>{step.label}</div>
            <div style={{ fontFamily:T.mono, fontSize:10, color:T.t3, marginTop:2 }}>{step.heure}</div>
          </div>
        </div>
      ))}
    </div>
  );
}
